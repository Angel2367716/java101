int i = 1;
while (i < 5 ) {
   System.out.print(i + "" );
   i++;
}
i = 1
do {
   System.out.print(i + "" );
   i++;
} while (i<5);
while (i < 5 ) {
   System.out.print(i + "" );
   i++;
}
do {
   System.out.print(i + "" );
   i++;
} while (i<5);
System.out.println("a do loop will be executed at least once, even if the condition is met. A while loop will not execute if the condition is not met");
